import streamlit as st
import pandas as pd
import io

# Configuración visual de la aplicación con la nueva identidad
st.set_page_config(page_title="Informe Concentrado Semanal", page_icon="📋", layout="wide")
st.title("📋 Informe Concentrado Semanal - Tienda Curicó")
st.markdown("Herramienta automática para procesar consumos de concentrados, ordenando por fecha y rebajando el lote más antiguo.")

# Área de carga del archivo Excel
archivo_ax = st.file_uploader("Sube el archivo Excel exportado de AX365 (.xlsx)", type=["xlsx"])

if archivo_ax is not None:
    try:
        excel_book = pd.ExcelFile(archivo_ax)
        hojas = excel_book.sheet_names
        
        st.sidebar.header("⚙️ Configuración de Hojas")
        hoja_inventario = st.sidebar.selectbox("Pestaña de Inventario (Base de datos):", hojas, index=0)
        hoja_consumos = st.sidebar.selectbox("Pestaña de Consumos Semanales:", hojas, index=1 if len(hojas) > 1 else 0)
        
        with st.spinner("🔍 Analizando códigos, fechas y lotes para el Informe..."):
            df_inv = pd.read_excel(archivo_ax, sheet_name=hoja_inventario)
            df_cons = pd.read_excel(archivo_ax, sheet_name=hoja_consumos)
            
            # --- DETECCIÓN INTELIGENTE DE COLUMNAS ---
            def buscar_columna(df, palabras_clave):
                for col in df.columns:
                    if any(p in str(col).lower() for p in palabras_clave):
                        return col
                return None

            col_codigo_inv = buscar_columna(df_inv, ['código ax', 'codigo ax', 'artículo', 'articulo', 'item', 'código de producto', 'codigo de producto', 'concentrado'])
            col_codigo_cons = buscar_columna(df_cons, ['código ax', 'codigo ax', 'artículo', 'articulo', 'item', 'código de producto', 'codigo de producto', 'concentrado'])
            col_lote = buscar_columna(df_inv, ['lote', 'batch', 'número de lote', 'numero de lote', 'lotes consumidos'])
            col_fecha = buscar_columna(df_inv, ['fecha', 'date', 'vencimiento', 'ingreso', 'creación'])
            col_stock = buscar_columna(df_inv, ['stock', 'inventario', 'físico', 'physical', 'on-hand', 'cantidad'])
            col_gasto = buscar_columna(df_cons, ['consumo', 'gasto', 'cantidad', 'quantity'])

            # Validaciones de seguridad en pantalla
            if not col_codigo_inv or not col_codigo_cons:
                st.error("🚨 No se encontró la columna de 'Código de Producto' o 'Código AX' en tus pestañas.")
                st.stop()
            if not col_fecha:
                st.error("🚨 No se encontró una columna de 'Fecha' (ingreso o vencimiento) en la pestaña de inventario.")
                st.stop()
            if not col_lote:
                st.error("🚨 No se encontró la columna de 'Lote' en la pestaña de inventario.")
                st.stop()
            if not col_stock or not col_gasto:
                st.error("🚨 No se encontraron las columnas numéricas de Stock o Consumo.")
                st.stop()

            # --- LIMPIEZA Y SANEAMIENTO DE FORMATOS ERP ---
            df_inv = df_inv.dropna(subset=[col_codigo_inv]).copy()
            df_cons = df_cons.dropna(subset=[col_codigo_cons]).copy()
            
            # Convertir códigos y lotes a texto limpio
            df_inv[col_codigo_inv] = df_inv[col_codigo_inv].astype(str).str.strip().str.replace(r'\.0$', '', regex=True)
            df_cons[col_codigo_cons] = df_cons[col_codigo_cons].astype(str).str.strip().str.replace(r'\.0$', '', regex=True)
            df_inv[col_lote] = df_inv[col_lote].astype(str).str.strip()
            
            # Convertir fechas de forma segura
            df_inv[col_fecha] = pd.to_datetime(df_inv[col_fecha], errors='coerce')
            
            # Convertir valores de stock y consumo a números (limpiando errores #VALOR!)
            df_inv[col_stock] = pd.to_numeric(df_inv[col_stock], errors='coerce').fillna(0)
            df_cons[col_gasto] = pd.to_numeric(df_cons[col_gasto], errors='coerce').fillna(0)
            
            # --- ORDENAMIENTO CRONOLÓGICO CLAVE (MÉTODO FIFO) ---
            df_inv = df_inv.sort_values(by=[col_codigo_inv, col_fecha], ascending=[True, True]).reset_index(drop=True)
            
            reporte_bajas = []
            resumen_por_codigo = {}
            alertas_quiebre = []
            
            # --- PROCESAMIENTO DE DESCUENTOS ---
            for _, fila_c in df_cons.iterrows():
                cod = fila_c[col_codigo_cons]
                gasto = fila_c[col_gasto]
                
                if gasto <= 0:
                    continue
                
                resumen_por_codigo[cod] = resumen_por_codigo.get(cod, 0) + gasto
                
                lotes_producto = df_inv[df_inv[col_codigo_inv] == cod].copy()
                
                if lotes_producto.empty:
                    continue
                
                for idx, lote in lotes_producto.iterrows():
                    if gasto <= 0:
                        break
                    
                    stock_lote_actual = df_inv.at[idx, col_stock]
                    
                    if stock_lote_actual > 0:
                        descuento = min(stock_lote_actual, gasto)
                        df_inv.at[idx, col_stock] -= descuento
                        gasto -= descuento
                        
                        fecha_str = lote[col_fecha].strftime('%Y-%m-%d') if pd.notnull(lote[col_fecha]) else "Sin Fecha"
                        
                        reporte_bajas.append({
                            'Código Producto': cod,
                            'Fecha Lote': fecha_str,
                            'Lote Consumido': lote[col_lote],
                            'Cantidad Extraída del Lote': round(descuento, 4),
                            'Stock Restante en Lote': round(df_inv.at[idx, col_stock], 4)
                        })
                
                if gasto > 0:
                    alertas_quiebre.append(f"🚨 Alerta: El producto {cod} tiene un quiebre. Faltaron {gasto:.3f} unidades por rebajar.")
            
            st.success("✨ ¡Análisis completado! Tu Informe Concentrado Semanal está listo.")
            
            # Crear DataFrames de resumen para la visualización
            df_resumen_totales = pd.DataFrame(list(resumen_por_codigo.items()), columns=['Código Producto', 'Total Consumido Semanal'])
            df_reporte_bajas = pd.DataFrame(reporte_bajas)
            
            # Guardado del archivo Excel en memoria con nombres de pestañas adaptados
            salida_excel = io.BytesIO()
            with pd.ExcelWriter(salida_excel, engine='openpyxl') as writer:
                df_inv.to_excel(writer, sheet_name='Inventario Actualizado', index=False)
                if not df_reporte_bajas.empty:
                    df_reporte_bajas.to_excel(writer, sheet_name='Lotes Consumidos Semanal', index=False)
                    df_resumen_totales.to_excel(writer, sheet_name='Consumo Total por Código', index=False)
            salida_excel.seek(0)
            
            # Botón de descarga con el nuevo nombre de archivo oficial
            st.download_button(
                label="📥 Descargar Informe Concentrado Semanal (Excel)",
                data=salida_excel,
                file_name="Informe_Concentrado_Semanal.xlsx",
                mime="application/vnd.openxmlformats-officedocument.spreadsheetml.sheet",
                type="primary"
            )
            
            if alertas_quiebre:
                st.subheader("⚠️ Descalces Registrados")
                for al in alertas_quiebre:
                    st.warning(al)
                    
            # --- PESTAÑAS INFORMATIVAS ---
            tab1, tab2, tab3 = st.tabs([
                "📊 Detalle de Lotes Consumidos", 
                "📉 Consumo Total por Artículo", 
                "🗄️ Inventario Final Corregido"
            ])
            
            with tab1:
                if not df_reporte_bajas.empty:
                    st.markdown("### Lotes específicos que registraron salidas:")
                    st.dataframe(df_reporte_bajas, use_container_width=True)
                else:
                    st.info("No se registraron movimientos en esta carga.")
                    
            with tab2:
                if not df_resumen_totales.empty:
                    st.markdown("### Resumen acumulado de consumo semanal por código:")
                    st.dataframe(df_resumen_totales, use_container_width=True)
                    
            with tab3:
                st.markdown("### Maestro de inventario con stock actualizado por lote:")
                st.dataframe(df_inv, use_container_width=True)
                
    except Exception as e:
        st.error(f"❌ Error al generar el informe cronológico: {e}. Confirma la estructura del archivo.")
